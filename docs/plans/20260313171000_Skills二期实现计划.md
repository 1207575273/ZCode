# Skills 二期 — 外置插件加载与 Hook 系统实现计划

> **For agentic workers:** Use superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 Skills 一期基础上新增外置插件包加载（带命名空间）、Skill 支撑文件元数据、Hook 事件系统（SessionStart / PreToolUse / PostToolUse）

**Architecture:** SkillStore 扩展四源扫描（builtin → plugin → user → project），插件 skill 自动加 `<插件名>:` 命名空间。新增 HookManager 模块处理三层 hooks.json 发现与合并，HookRunner 负责子进程执行。Hook 通过环境变量+stdin 传入上下文，stdout JSON 返回结果。

**Tech Stack:** TypeScript strict, Node.js child_process, Vitest, fast-glob

```typescript
import { basename } from 'node:path'

/** 用户级插件目录 */
function userPluginsDir(): string {
  return join(homedir(), '.ccode', 'plugins')
}

/** 项目级插件目录 */
function projectPluginsDir(): string {
  return join(process.cwd(), '.ccode', 'plugins')
}

// 在 SkillStore 类中新增：

/** 扫描所有插件包 */
async #scanPlugins(): Promise<SkillMetadata[]> {
  const results: SkillMetadata[] = []
  const dirs = [userPluginsDir(), projectPluginsDir()]

  for (const baseDir of dirs) {
    try {
      // 先发现所有插件包目录
      const pattern = baseDir.replace(/\\/g, '/') + '/*/'
      const pluginDirs = await fg(pattern, { onlyDirectories: true, absolute: true })

      for (const pluginDir of pluginDirs) {
        const pluginName = basename(pluginDir.replace(/\/$/, ''))
        const skillsDir = join(pluginDir, 'skills')
        const skills = await this.#scanDir(skillsDir, 'plugin')

        for (const skill of skills) {
          skill.name = `${pluginName}:${skill.name}`
          skill.pluginName = pluginName
          results.push(skill)
        }
      }
    } catch {
      // 目录不存在，正常
    }
  }
  return results
}
```

修改 `discover()` 方法，增加 plugin 层：

```typescript
async discover(): Promise<SkillMetadata[]> {
  if (this.#discovered) return [...this.#skills.values()]
  this.#discovered = true

  const [builtinSkills, pluginSkills, userSkills, projectSkills] = await Promise.all([
    this.#scanDir(builtinSkillsDir(), 'builtin'),
    this.#scanPlugins(),
    this.#scanDir(userSkillsDir(), 'user'),
    this.#scanDir(projectSkillsDir(), 'project'),
  ])

  for (const meta of [...builtinSkills, ...pluginSkills, ...userSkills, ...projectSkills]) {
    this.#skills.set(meta.name, meta)
  }

  return [...this.#skills.values()]
}
```

```typescript
/** 列举 skill 目录下的支撑文件（排除 SKILL.md、隐藏文件、node_modules） */
async getSupportingFiles(name: string): Promise<string[]> {
  const meta = this.#skills.get(name)
  if (!meta) return []

  const skillDir = dirname(meta.filePath).replace(/\\/g, '/')
  const pattern = skillDir + '/**/*'
  try {
    const files = await fg(pattern, {
      absolute: true,
      ignore: ['**/node_modules/**', '**/.*'],
      onlyFiles: true,
    })
    // 过滤掉 SKILL.md 本身，返回相对路径
    return files
      .filter(f => !f.endsWith('/SKILL.md') && !f.endsWith('\\SKILL.md'))
      .map(f => {
        const rel = f.replace(skillDir + '/', '').replace(skillDir + '\\', '')
        return rel.replace(/\\/g, '/')
      })
  } catch {
    return []
  }
}
```

```typescript
async execute(args: Record<string, unknown>, _ctx: ToolContext): Promise<ToolResult> {
  const name = args['name']
  if (typeof name !== 'string' || !name.trim()) {
    return { success: false, output: '', error: 'Missing required parameter: name' }
  }

  const trimmedName = name.trim()
  const content = await this.#store.getContent(trimmedName)
  if (!content) {
    const available = this.#store.getAll().map(s => s.name).join(', ')
    return {
      success: false,
      output: '',
      error: `Skill "${trimmedName}" not found. Available skills: ${available || 'none'}`,
    }
  }

  // 附加支撑文件元数据
  const supportingFiles = await this.#store.getSupportingFiles(trimmedName)
  const meta = this.#store.get(trimmedName)

  let output = ''
  if (supportingFiles.length > 0 && meta) {
    const baseDir = dirname(meta.filePath).replace(/\\/g, '/')
    const fileList = supportingFiles.map(f => `- ${f}`).join('\n')
    output += `<skill-context name="${trimmedName}" base-dir="${baseDir}">\n`
    output += `Supporting files (use Read tool to access):\n${fileList}\n`
    output += `</skill-context>\n\n`
  }
  output += content

  return { success: true, output }
}
```

```typescript
// src/hooks/hook-runner.ts

import { spawn } from 'node:child_process'

export interface RunOptions {
  command: string
  cwd: string
  env: Record<string, string>
  timeout: number
  stdin?: string
}

export class HookRunner {
  /** 执行 hook 命令，返回解析后的 JSON 或 null */
  async run(opts: RunOptions): Promise<Record<string, unknown> | null> {
    return new Promise((resolve) => {
      const { command, cwd, env, timeout, stdin } = opts

      const child = spawn('bash', ['-c', command], {
        cwd,
        env: { ...process.env, ...env },
        stdio: ['pipe', 'pipe', 'pipe'],
        timeout,
      })

      let stdout = ''
      let settled = false

      const settle = (value: Record<string, unknown> | null) => {
        if (settled) return
        settled = true
        resolve(value)
      }

      child.stdout.on('data', (chunk: Buffer) => { stdout += chunk.toString() })
      child.on('error', () => settle(null))
      child.on('close', (code) => {
        if (code !== 0) return settle(null)
        try {
          const parsed = JSON.parse(stdout.trim())
          if (typeof parsed === 'object' && parsed !== null) {
            settle(parsed as Record<string, unknown>)
          } else {
            settle(null)
          }
        } catch {
          settle(null)
        }
      })

      // 写入 stdin
      if (stdin) {
        child.stdin.write(stdin)
      }
      child.stdin.end()
    })
  }
}
```

```typescript
// src/hooks/hook-manager.ts

import { readFile } from 'node:fs/promises'
import { dirname } from 'node:path'
import { HookRunner } from './hook-runner.js'
import type {
  HookEventType, HooksConfig, ResolvedHookEntry, HookContext,
} from './types.js'

export class HookManager {
  #entries: ResolvedHookEntry[] = []
  readonly #runner = new HookRunner()

  /** 从一个 hooks.json 文件加载 hook 定义 */
  async discoverFromFile(
    filePath: string,
    source: ResolvedHookEntry['source'],
    pluginName?: string,
  ): Promise<void> {
    try {
      const raw = await readFile(filePath, 'utf-8')
      const config: HooksConfig = JSON.parse(raw)
      const cwd = dirname(filePath)

      for (const [eventStr, rules] of Object.entries(config.hooks)) {
        const event = eventStr as HookEventType
        if (!rules) continue
        for (const rule of rules) {
          const matcher = new RegExp(rule.matcher)
          for (const action of rule.hooks) {
            this.#entries.push({ source, pluginName, event, matcher, action, cwd })
          }
        }
      }
    } catch {
      // 文件不存在或 JSON 解析失败，静默跳过
    }
  }

  /** 获取指定事件的全部 hook */
  getHooks(event: HookEventType): ResolvedHookEntry[] {
    return this.#entries.filter(e => e.event === event)
  }

  /** 获取指定事件中匹配 trigger 的 hook */
  getMatchedHooks(event: HookEventType, trigger: string): ResolvedHookEntry[] {
    return this.#entries.filter(e => e.event === event && e.matcher.test(trigger))
  }

  /** 执行指定事件的所有匹配 hook，返回结果列表 */
  async run(event: HookEventType, ctx: HookContext): Promise<Array<Record<string, unknown> | null>> {
    const matched = this.getMatchedHooks(event, ctx.trigger)
    const results: Array<Record<string, unknown> | null> = []

    for (const entry of matched) {
      const env: Record<string, string> = { ...ctx.env }
      if (entry.pluginName) {
        // 插件 hook：设置 CCODE_PLUGIN_ROOT
        const pluginRoot = dirname(entry.cwd)
        env['CCODE_PLUGIN_ROOT'] = pluginRoot
      }
      const result = await this.#runner.run({
        command: entry.action.command,
        cwd: entry.cwd,
        env,
        timeout: entry.action.timeout ?? 10000,
        stdin: ctx.stdin,
      })
      results.push(result)
    }

    return results
  }
}
```

```typescript
export const hookManager = new HookManager()

/** 发现所有 hooks（从插件包 + 项目级 + 用户级） */
export async function ensureHooksDiscovered(): Promise<void> {

  // 3. 用户级

}

/** 执行 SessionStart hooks，返回合并的 additionalContext */
export async function runSessionStartHooks(trigger: string): Promise<string> {
  const results = await hookManager.run('SessionStart', {
    trigger,
    env: {
      CCODE_CWD: process.cwd(),
      CCODE_TRIGGER: trigger,
    },
  })

  const contexts: string[] = []
  for (const r of results) {
    if (!r) continue
    // 兼容 Claude Code 格式
    const ctx = (r['hookSpecificOutput'] as Record<string, unknown>)?.['additionalContext']
      ?? r['additionalContext']
      ?? r['additional_context']

      contexts.push(ctx)
    }
  }
  return contexts.join('\n\n')
```
