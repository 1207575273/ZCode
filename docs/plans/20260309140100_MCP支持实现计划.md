```typescript
// src/config/mcp-config.ts
import { readFileSync, existsSync } from 'node:fs'
import { join } from 'node:path'
import { homedir } from 'node:os'

export interface McpServerConfig {
  /** stdio 传输：可执行命令 */
  command?: string
  args?: string[]
  env?: Record<string, string>
  /** 传输类型：不写默认 stdio（有 command 时），sse 或 streamable-http */
  type?: 'stdio' | 'sse' | 'streamable-http'
  /** SSE / streamable-http 传输的 URL */
  url?: string
}

export interface McpConfig {
  mcpServers: Record<string, McpServerConfig>
}

const EMPTY_CONFIG: McpConfig = { mcpServers: {} }

/** 默认路径 ~/.ccode/mcp.json */
export const DEFAULT_MCP_CONFIG_PATH = join(homedir(), '.ccode', 'mcp.json')

/** 加载 mcp.json，不存在或格式错误时返回空配置 */
export function loadMcpConfig(configPath: string = DEFAULT_MCP_CONFIG_PATH): McpConfig {
  if (!existsSync(configPath)) return { ...EMPTY_CONFIG, mcpServers: {} }
  try {
    const raw = readFileSync(configPath, 'utf-8')
    const parsed = JSON.parse(raw) as Record<string, unknown>
    const mcpServers = parsed['mcpServers']
    if (mcpServers == null || typeof mcpServers !== 'object') {
      return { ...EMPTY_CONFIG, mcpServers: {} }
    }
    return { mcpServers: mcpServers as Record<string, McpServerConfig> }
  } catch {
    return { ...EMPTY_CONFIG, mcpServers: {} }
  }
}
```

```typescript
// src/mcp/mcp-tool.ts
import type { Client } from '@modelcontextprotocol/sdk/client'
import type { Tool, ToolContext, ToolResult } from '@tools/types.js'

/** MCP Server 返回的工具定义 */
export interface McpToolDefinition {
  name: string
  description?: string
  inputSchema: Record<string, unknown>
}

/**
 * MCP 工具适配器。
 * 将 MCP Server 提供的工具包装为 cCli 内部 Tool 接口。
 * 命名规则：mcp__<serverName>__<toolName>
 */
export class McpTool implements Tool {
  readonly name: string
  readonly description: string
  readonly parameters: Record<string, unknown>
  readonly dangerous = true

  constructor(
    private readonly client: Client,
    private readonly serverName: string,
    private readonly toolDef: McpToolDefinition,
  ) {
    this.name = `mcp__${serverName}__${toolDef.name}`
    this.description = toolDef.description ?? ''
    this.parameters = toolDef.inputSchema
  }

  async execute(args: Record<string, unknown>, _ctx: ToolContext): Promise<ToolResult> {
    try {
      const result = await this.client.callTool({
        name: this.toolDef.name,
        arguments: args,
      })

      // 提取文本内容
      const contents = result.content as Array<{ type: string; text?: string }> | undefined
      const text = contents
        ?.filter(c => c.type === 'text' && c.text != null)
        .map(c => c.text)
        .join('\n') ?? ''

      if (result.isError) {
        return { success: false, output: '', error: text || 'MCP tool returned error' }
      }
      return { success: true, output: text || '(no output)' }
    } catch (err) {
      return { success: false, output: '', error: err instanceof Error ? err.message : String(err) }
    }
  }
}
```

```typescript
// src/mcp/mcp-manager.ts
import { Client } from '@modelcontextprotocol/sdk/client'
import { StdioClientTransport } from '@modelcontextprotocol/sdk/client/stdio.js'
import { SSEClientTransport } from '@modelcontextprotocol/sdk/client/sse.js'
import { StreamableHTTPClientTransport } from '@modelcontextprotocol/sdk/client/streamableHttp.js'
import type { McpConfig, McpServerConfig } from '@config/mcp-config.js'
import { McpTool } from './mcp-tool.js'
import type { McpToolDefinition } from './mcp-tool.js'

const CLIENT_INFO = { name: 'ccode', version: '0.1.0' }
const CONNECT_TIMEOUT = 30_000

interface ConnectedServer {
  name: string
  client: Client
  tools: McpTool[]
}

/**
 * MCP Server 连接管理器。
 * 负责：读取配置 → 连接 Server → 收集工具 → 断开连接。
 */
export class McpManager {
  private readonly servers: ConnectedServer[] = []

  constructor(private readonly config: McpConfig) {}

  /** 连接所有配置的 MCP Server，单个失败不影响其余 */
  async connectAll(): Promise<void> {
    const entries = Object.entries(this.config.mcpServers)
    for (const [name, serverConfig] of entries) {
      try {
        const client = new Client(CLIENT_INFO)
        const transport = this.createTransport(serverConfig)
        await client.connect(transport)

        const { tools: toolDefs } = await client.listTools()
        const tools = toolDefs.map(td => new McpTool(
          client,
          name,
          { name: td.name, description: td.description, inputSchema: td.inputSchema } as McpToolDefinition,
        ))

        this.servers.push({ name, client, tools })
      } catch (err) {
        const msg = err instanceof Error ? err.message : String(err)
        console.error(`[MCP] Failed to connect "${name}": ${msg}`)
      }
    }
  }

  /** 获取所有已连接 Server 的工具列表 */
  getTools(): McpTool[] {
    return this.servers.flatMap(s => s.tools)
  }

  /** 断开所有连接 */
  async disconnectAll(): Promise<void> {
    for (const server of this.servers) {
      try {
        await server.client.close()
      } catch {
        // 静默处理断开失败
      }
    }
    this.servers.length = 0
  }

  private createTransport(config: McpServerConfig) {
    // 有 command → stdio
    if (config.command != null) {
      return new StdioClientTransport({
        command: config.command,
        args: config.args,
        env: config.env != null ? { ...process.env, ...config.env } as Record<string, string> : undefined,
      })
    }
    // type + url → SSE 或 streamable-http
    if (config.url != null) {
      const url = new URL(config.url)
      if (config.type === 'sse') {
        return new SSEClientTransport(url)
      }
      // streamable-http 或默认
      return new StreamableHTTPClientTransport(url)
    }
    throw new Error('Invalid MCP server config: need command or url')
  }
}
```

```typescript
import { loadMcpConfig } from '@config/mcp-config.js'
import { McpManager } from '@mcp/mcp-manager.js'

// 模块级变量（跨 submit 保持连接）
let mcpManager: McpManager | null = null

/** 初始化 MCP（仅执行一次） */
async function ensureMcpConnected(registry: ToolRegistry): Promise<void> {
  if (mcpManager != null) {
    // 已连接，只注册工具
    for (const tool of mcpManager.getTools()) {
      registry.register(tool)
    }
    return
  }
  const config = loadMcpConfig()
  if (Object.keys(config.mcpServers).length === 0) return

  mcpManager = new McpManager(config)
  await mcpManager.connectAll()
  for (const tool of mcpManager.getTools()) {
    registry.register(tool)
  }
}

// 修改 submit 函数中的 registry 构建部分：
// 原来：const registry = buildRegistry()
```

```typescript
    const argsStr = Object.keys(args).length > 0 ? JSON.stringify(args, null, 2) : ''
    return `[${server}] ${tool}\n${argsStr}`
  }
  return JSON.stringify(args)
```
