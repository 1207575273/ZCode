# F8 会话持久化与恢复 实现计划

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** 实现 JSONL 事件流自动保存 + ResumePanel 交互式恢复，让用户关闭 ZCli 后可恢复历史对话。

**Architecture:** SessionStore 负责 JSONL 文件读写（append-only），useChat 在每个关键事件后调用 append。ResumePanel 作为独立全屏面板（与 ModelPicker/McpStatusView 同模式），展示最近会话供用户选择恢复。CLI 入口支持 `--resume` 参数。

**Tech Stack:** Node.js `fs/promises` + `uuid`(v7) + Ink React 组件

```typescript
// src/persistence/session-utils.ts
import { v7 as uuidv7, v4 as uuidv4 } from 'uuid'
import { execSync } from 'node:child_process'

/** 将 cwd 路径转为文件系统安全的目录名（与 Claude Code 一致） */
export function toProjectSlug(cwd: string): string {
  return cwd.replace(/[/\\:]/g, '-')
}

/** 生成 UUIDv7（时间有序） */
export function generateSessionId(): string {
  return uuidv7()
}

/** 生成普通 UUIDv4（事件 ID） */
export function generateEventId(): string {
  return uuidv4()
}

/** 格式化会话文件名: YYYYMMDDHHMMSSMMM_<sessionId>.jsonl */
export function formatSessionFilename(sessionId: string, date: Date = new Date()): string {
  const y = date.getUTCFullYear()
  const mo = String(date.getUTCMonth() + 1).padStart(2, '0')
  const d = String(date.getUTCDate()).padStart(2, '0')
  const h = String(date.getUTCHours()).padStart(2, '0')
  const mi = String(date.getUTCMinutes()).padStart(2, '0')
  const s = String(date.getUTCSeconds()).padStart(2, '0')
  const ms = String(date.getUTCMilliseconds()).padStart(3, '0')
  return `${y}${mo}${d}${h}${mi}${s}${ms}_${sessionId}.jsonl`
}

/** 从文件名中提取 sessionId */
export function extractSessionId(filename: string): string {
  // 20260309153022317_<uuid>.jsonl → <uuid>
  const match = filename.match(/^\d{17}_(.+)\.jsonl$/)
  return match?.[1] ?? filename.replace('.jsonl', '')
}

/** 获取当前 git 分支名，失败返回 'unknown' */
export function getGitBranch(cwd: string): string {
  try {
    return execSync('git rev-parse --abbrev-ref HEAD', { cwd, encoding: 'utf-8', timeout: 3000 }).trim()
  } catch {
    return 'unknown'
  }
}
```

```typescript
// src/persistence/session-store.ts
import { readFileSync, writeFileSync, appendFileSync, mkdirSync, readdirSync, statSync } from 'node:fs'
import { join } from 'node:path'
import type { SessionEvent, SessionSnapshot, SessionSummary } from './session-types.js'
import {
  toProjectSlug,
  generateSessionId,
  generateEventId,
  formatSessionFilename,
  extractSessionId,
  getGitBranch,
} from './session-utils.js'

/**
 * SessionStore — JSONL 事件流的读写、列出、清理。
 *
 * 目录结构:
 *   <baseDir>/<projectSlug>/<YYYYMMDDHHMMSSMMM>_<sessionId>.jsonl
 */
export class SessionStore {
  readonly #baseDir: string
  /** sessionId → filePath 缓存，避免每次 append 时重新扫描 */
  readonly #pathCache = new Map<string, string>()

  constructor(baseDir: string) {
    this.#baseDir = baseDir
  }

  /** 创建新会话，写入 session_start 事件，返回 sessionId */
  create(cwd: string, provider: string, model: string): string {
    const sessionId = generateSessionId()
    const slug = toProjectSlug(cwd)
    const dir = join(this.#baseDir, slug)
    mkdirSync(dir, { recursive: true })

    const filename = formatSessionFilename(sessionId)
    const filePath = join(dir, filename)
    this.#pathCache.set(sessionId, filePath)

    const event: SessionEvent = {
      sessionId,
      type: 'session_start',
      timestamp: new Date().toISOString(),
      uuid: generateEventId(),
      parentUuid: null,
      cwd,
      gitBranch: getGitBranch(cwd),
      provider,
      model,
    }
    writeFileSync(filePath, JSON.stringify(event) + '\n', 'utf-8')
    return sessionId
  }

  /** 追加事件到 JSONL 文件 */
  append(sessionId: string, event: SessionEvent): void {
    const filePath = this.#resolveFilePath(sessionId)
    if (!filePath) return
    appendFileSync(filePath, JSON.stringify(event) + '\n', 'utf-8')
  }

  /** 从 JSONL 回放，提取 user/assistant 消息重建 ChatMessage[] */
  loadMessages(sessionId: string): SessionSnapshot {
    const filePath = this.#resolveFilePath(sessionId)
    if (!filePath) {
      throw new Error(`Session not found: ${sessionId}`)
    }

    const lines = readFileSync(filePath, 'utf-8').split('\n').filter(Boolean)
    let provider = ''
    let model = ''
    let cwd = ''
    const messages: SessionSnapshot['messages'] = []

    for (const line of lines) {
      let event: SessionEvent
      try {
        event = JSON.parse(line) as SessionEvent
      } catch {
        continue
      }

      if (event.type === 'session_start' || event.type === 'session_resume') {
        provider = event.provider ?? provider
        model = event.model ?? model
        cwd = event.cwd ?? cwd
      }

      if (
        (event.type === 'user' || event.type === 'assistant') &&
        event.message &&
        typeof event.message.content === 'string'
      ) {
        messages.push({
          id: event.uuid,
          role: event.type as 'user' | 'assistant',
          content: event.message.content,
        })
      }
    }

    return { sessionId, provider, model, cwd, messages }
  }

  /** 列出会话（按 updatedAt 降序） */
  list(options?: { projectSlug?: string; limit?: number }): SessionSummary[] {
    const limit = options?.limit ?? 10
    const summaries: SessionSummary[] = []

    const slugs = options?.projectSlug
      ? [options.projectSlug]
      : this.#listProjectSlugs()

    for (const slug of slugs) {
      const dir = join(this.#baseDir, slug)
      let files: string[]
      try {
        files = readdirSync(dir).filter(f => f.endsWith('.jsonl')).sort().reverse()
      } catch {
        continue
      }

      for (const file of files) {
        const filePath = join(dir, file)
        const sessionId = extractSessionId(file)
        this.#pathCache.set(sessionId, filePath)

        let fileSize: number
        try {
          fileSize = statSync(filePath).size
        } catch {
          continue
        }

        // 读取前几行提取摘要信息
        const { firstMessage, updatedAt, gitBranch } = this.#extractSummary(filePath)
        summaries.push({
          sessionId,
          projectSlug: slug,
          firstMessage,
          updatedAt,
          gitBranch,
          fileSize,
          filePath,
        })
      }
    }

    // 按 updatedAt 降序排列
    summaries.sort((a, b) => b.updatedAt.localeCompare(a.updatedAt))
    return summaries.slice(0, limit)
  }

  /** 清理过期会话 */
  cleanup(retentionDays: number = 30): void {
    const cutoff = Date.now() - retentionDays * 24 * 60 * 60 * 1000
    for (const slug of this.#listProjectSlugs()) {
      const dir = join(this.#baseDir, slug)
      let files: string[]
      try {
        files = readdirSync(dir).filter(f => f.endsWith('.jsonl'))
      } catch {
        continue
      }
      for (const file of files) {
        const filePath = join(dir, file)
        try {
          const stat = statSync(filePath)
          if (stat.mtimeMs < cutoff) {
            const { unlinkSync } = require('node:fs')
            unlinkSync(filePath)
          }
        } catch {
          // 静默忽略
        }
      }
    }
  }

  /** 从 JSONL 提取摘要信息（首条 user 消息 + 最后时间戳 + git branch） */
  #extractSummary(filePath: string): { firstMessage: string; updatedAt: string; gitBranch: string } {
    let firstMessage = '(empty session)'
    let updatedAt = ''
    let gitBranch = 'unknown'

    try {
      const content = readFileSync(filePath, 'utf-8')
      const lines = content.split('\n').filter(Boolean)

      for (const line of lines) {
        try {
          const event = JSON.parse(line) as SessionEvent
          if (!updatedAt || event.timestamp > updatedAt) {
            updatedAt = event.timestamp
          }
          if (event.gitBranch) {
            gitBranch = event.gitBranch
          }
          if (event.type === 'user' && event.message && typeof event.message.content === 'string' && firstMessage === '(empty session)') {
            firstMessage = event.message.content.slice(0, 80)
          }
        } catch {
          continue
        }
      }
    } catch {
      // 文件读取失败
    }

    return { firstMessage, updatedAt: updatedAt || new Date().toISOString(), gitBranch }
  }

  /** 解析 sessionId 对应的 JSONL 文件路径 */
  #resolveFilePath(sessionId: string): string | undefined {
    if (this.#pathCache.has(sessionId)) return this.#pathCache.get(sessionId)

    // 遍历所有项目目录查找
    for (const slug of this.#listProjectSlugs()) {
      const dir = join(this.#baseDir, slug)
      try {
        const files = readdirSync(dir)
        const match = files.find(f => f.includes(sessionId) && f.endsWith('.jsonl'))
        if (match) {
          const filePath = join(dir, match)
          this.#pathCache.set(sessionId, filePath)
          return filePath
        }
      } catch {
        continue
      }
    }
    return undefined
  }

  /** 列出所有项目 slug 目录 */
  #listProjectSlugs(): string[] {
    try {
      return readdirSync(this.#baseDir, { withFileTypes: true })
        .filter(d => d.isDirectory())
        .map(d => d.name)
    } catch {
      return []
    }
  }
}
```

**注意：** cleanup 方法中的 `require('node:fs')` 应改为顶部导入的 `unlinkSync`。在实现时将 `unlinkSync` 加入顶部 import。

## Task 7: ResumePanel UI 组件

**Files:**

- Modify: `zCli/src/ui/App.tsx` — 互斥渲染 ResumePanel

**Step 1: 实现 ResumePanel**

组件功能：
- 展示会话列表（SessionSummary[]），当前项目 / 所有项目可切换
- `↑↓` 导航，`Enter` 选中，`Ctrl+A` 切换范围，输入文字搜索，`Esc` 关闭
- 与 ModelPicker 相同的 `useInput` + `useCallback` 稳定 handler 模式

```typescript
// src/ui/ResumePanel.tsx
import React, { useState, useCallback, useMemo } from 'react'
import { Box, Text, useInput } from 'ink'
import type { Key } from 'ink'
import type { SessionSummary } from '@persistence/session-types.js'

interface ResumePanelProps {
  /** 当前项目的会话列表 */
  currentProjectSessions: SessionSummary[]
  /** 所有项目的会话列表 */
  allSessions: SessionSummary[]
  onSelect: (sessionId: string) => void
  onClose: () => void
}

/** 格式化相对时间 */
function timeAgo(isoString: string): string {
  const diff = Date.now() - new Date(isoString).getTime()
  const seconds = Math.floor(diff / 1000)
  if (seconds < 60) return `${seconds}s ago`
  const minutes = Math.floor(seconds / 60)
  if (minutes < 60) return `${minutes}m ago`
  const hours = Math.floor(minutes / 60)
  if (hours < 24) return `${hours}h ago`
  const days = Math.floor(hours / 24)
  return `${days}d ago`
}

/** 格式化文件大小 */
function formatSize(bytes: number): string {
  if (bytes < 1024) return `${bytes} B`
  const kb = bytes / 1024
  if (kb < 1024) return `${kb.toFixed(1)}KB`
  return `${(kb / 1024).toFixed(1)}MB`
}

export function ResumePanel({ currentProjectSessions, allSessions, onSelect, onClose }: ResumePanelProps) {
  const [showAll, setShowAll] = useState(false)
  const [selectedIndex, setSelectedIndex] = useState(0)
  const [search, setSearch] = useState('')

  const sessions = showAll ? allSessions : currentProjectSessions

  const filtered = useMemo(() => {
    if (!search) return sessions
    const q = search.toLowerCase()
    return sessions.filter(s => s.firstMessage.toLowerCase().includes(q))
  }, [sessions, search])

  useInput(useCallback((input: string, key: Key) => {
    if (key.escape) { onClose(); return }
    if (key.return && filtered.length > 0) {
      onSelect(filtered[selectedIndex]!.sessionId)
      return
    }
    if (key.upArrow) {
      setSelectedIndex(i => Math.max(0, i - 1))
      return
    }
    if (key.downArrow) {
      setSelectedIndex(i => Math.min(filtered.length - 1, i + 1))
      return
    }
    // Ctrl+A 切换
    if (input === '\x01') {
      setShowAll(prev => !prev)
      setSelectedIndex(0)
      return
    }
    // 普通字符 → 搜索
    if (input && !key.ctrl && !key.meta) {
      if (key.backspace || key.delete) {
        setSearch(prev => prev.slice(0, -1))
      } else if (input.length === 1 && input >= ' ') {
        setSearch(prev => prev + input)
      }
      setSelectedIndex(0)
    }
  }, [onClose, onSelect, filtered, selectedIndex]))

  if (filtered.length === 0) {
    return (
      <Box flexDirection="column" paddingX={2} paddingY={1}>
        <Text bold>Resume session</Text>
        <Text dimColor>{search ? '无匹配的会话' : '无历史会话'}</Text>
        <Box marginTop={1}>
          <Text dimColor>Esc 取消</Text>
        </Box>
      </Box>
    )
  }

  return (
    <Box flexDirection="column" paddingX={2} paddingY={1}>
      <Text bold>Resume session</Text>
      <Text dimColor>  {showAll ? '所有项目' : '当前项目'} · {filtered.length} sessions</Text>
      {search && <Text dimColor>  搜索: {search}</Text>}
      <Text> </Text>
      {filtered.map((s, i) => (
        <Box key={s.sessionId} flexDirection="column">
          <Text>
            {i === selectedIndex ? '❯ ' : '  '}
            <Text bold={i === selectedIndex}>
              {s.firstMessage.slice(0, 60)}
            </Text>
          </Text>
          <Text dimColor>
            {'  '}{timeAgo(s.updatedAt)} · {s.gitBranch} · {formatSize(s.fileSize)}
            {showAll ? ` · ${s.projectSlug}` : ''}
          </Text>
        </Box>
      ))}
      <Text> </Text>
      <Text dimColor>↑↓ 选择 · Enter 恢复 · Ctrl+A {showAll ? '当前项目' : '所有项目'} · Esc 取消</Text>
    </Box>
  )
}
```
