# F17 并行工具执行 Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** LLM 返回多个 tool_calls 时，安全工具并行执行以缩短耗时，危险工具串行执行保证权限交互正常。

**Architecture:** 在 AgentLoop 中将工具执行从 `for...of yield*` 串行循环改为「分组 → 并行安全组 → 串行危险组」策略。引入 `parallel-executor.ts` 封装并行调度逻辑，通过事件回调（而非 `yield*`）收集事件，完成后按顺序 yield 给消费者。AgentLoop 构造函数新增可选 `parallelTools` 开关（默认 true）。

**Tech Stack:** TypeScript, Promise.allSettled, Vitest

// src/core/parallel-executor.ts

/**
 * ParallelExecutor — 并行工具调度器。
 *
 * 将一组 toolCalls 按安全/危险分类，安全工具并行执行，危险工具串行执行。
 * 通过事件回调收集中间事件（tool_start / tool_done / permission_request 等），
 * 不直接 yield（因为 Promise.allSettled 无法与 AsyncGenerator yield* 共存）。
 */

import type { ToolRegistry } from '@tools/registry.js'
import type { ToolCallContent } from './types.js'

/** 分组结果 */
export interface ClassifiedToolCalls {
  safe: ToolCallContent[]
  dangerous: ToolCallContent[]
}

/**
 * 将 toolCalls 分为安全组和危险组。
 * 未在 registry 中注册的工具视为危险。
 */
export function classifyToolCalls(
  toolCalls: ToolCallContent[],
  registry: ToolRegistry,
): ClassifiedToolCalls {
  const safe: ToolCallContent[] = []
  const dangerous: ToolCallContent[] = []

  for (const tc of toolCalls) {
    if (registry.isDangerous(tc.toolName)) {
      dangerous.push(tc)
    } else {
      // isDangerous 对未注册工具返回 false（因为 get 返回 undefined，?.dangerous === true 为 false）
      // 但未注册工具应该视为危险 → 需要额外检查
      const known = registry.getAll().some(t => t.name === tc.toolName)
      if (known) {
        safe.push(tc)
      } else {
        dangerous.push(tc)
      }
    }
  }

  return { safe, dangerous }
}
```

// 追加到 src/core/parallel-executor.ts

import type { AgentEvent } from './agent-loop.js'
import type { ToolResult } from '@tools/types.js'

/** 最大并行工具数，防止同时打开过多文件/进程 */
const DEFAULT_MAX_PARALLEL = 5

/** 并行执行结果（带原始 toolCallId，用于 history 排序） */
export interface ParallelToolResult {
  toolCallId: string
  toolName: string
  result: ToolResult
  durationMs: number
  success: boolean
}

/**
 * 并行执行一组安全工具。
 *
 * - 通过 onEvent 回调收集 tool_start / tool_done 事件（不使用 yield*）
 * - 返回按原始 toolCalls 顺序排列的结果数组
 * - maxParallel 控制并发上限，超过时分批执行
 */
export async function executeSafeToolsInParallel(
  toolCalls: ToolCallContent[],
  registry: ToolRegistry,
  onEvent: (event: AgentEvent) => void,
  maxParallel: number = DEFAULT_MAX_PARALLEL,
): Promise<ParallelToolResult[]> {
  const results: ParallelToolResult[] = []
  const resultMap = new Map<string, ParallelToolResult>()

  // 分批执行（chunk by maxParallel）
  for (let i = 0; i < toolCalls.length; i += maxParallel) {
    const batch = toolCalls.slice(i, i + maxParallel)

    const promises = batch.map(async (tc) => {
      onEvent({ type: 'tool_start', toolName: tc.toolName, toolCallId: tc.toolCallId, args: tc.args })

      const start = Date.now()
      const result = await registry.execute(tc.toolName, tc.args, { cwd: process.cwd() })
      const durationMs = Date.now() - start
      const success = result.success

      const pr: ParallelToolResult = { toolCallId: tc.toolCallId, toolName: tc.toolName, result, durationMs, success }
      resultMap.set(tc.toolCallId, pr)

      const resultSummary = success
        ? (result.output.length > 200 ? result.output.slice(0, 200) + '...' : result.output)
        : (result.error ?? 'error')
      onEvent({ type: 'tool_done', toolName: tc.toolName, toolCallId: tc.toolCallId, durationMs, success, resultSummary })

      return pr
    })

    await Promise.allSettled(promises)
  }

  // 按原始 toolCalls 顺序返回结果
  for (const tc of toolCalls) {
    const pr = resultMap.get(tc.toolCallId)
    if (pr) results.push(pr)
  }

  return results
}
```

/**
 * 执行一组工具调用。
 *
 * parallelTools=true（默认）：安全工具并行 → 危险工具串行
 * parallelTools=false：全部串行（兼容模式）
 */
async *#executeToolCalls(
  toolCalls: ToolCallContent[],
  history: Message[],
): AsyncGenerator<AgentEvent> {
  // 串行模式（兼容）
  if (this.#config.parallelTools === false) {
    for (const tc of toolCalls) {
      yield* this.#executeOneTool(tc, history)
    }
    return
  }

  // 并行模式：分组
  const { safe, dangerous } = classifyToolCalls(toolCalls, this.#registry)

  // 1. 并行执行安全工具
  if (safe.length > 0) {
    const events: AgentEvent[] = []
    const results = await executeSafeToolsInParallel(
      safe,
      this.#registry,
      (e) => events.push(e),
      this.#config.maxParallelTools,
    )

    // yield 收集到的事件
    for (const e of events) {
      yield e
    }

    // 按原始顺序追加到 history
    for (const pr of results) {
      history.push({
        role: 'user',
        content: pr.success
          ? `[Tool ${pr.toolName} result]: ${pr.result.output}`
          : `[Tool ${pr.toolName} error]: ${pr.result.error ?? 'error'}`,
      })
    }
  }

  // 2. 串行执行危险工具（需要逐个权限确认）
  for (const tc of dangerous) {
    yield* this.#executeOneTool(tc, history)
  }
}

## 设计决策记录

| # | 决策 | 理由 |
|---|------|------|
| 1 | 新建 `parallel-executor.ts` 而非修改 `agent-loop.ts` | 关注点分离：并行调度逻辑独立于主循环，易于测试 |
| 2 | 使用事件回调而非 yield* | Promise.allSettled 内部无法 yield，回调收集后统一 yield |
| 3 | 未注册工具视为危险 | 安全优先，未知工具不应自动并行执行 |
| 4 | history 按原始 toolCalls 顺序追加 | 保证 LLM 看到确定性的结果顺序，避免不同运行产生不同上下文 |
| 5 | parallelTools 默认 true | 并行是明确的性能提升，默认启用；用户可通过配置关闭 |
| 6 | maxParallel 分批而非信号量 | 实现简单（slice + Promise.allSettled），对工具执行场景足够 |
| 7 | 先并行安全组，再串行危险组 | 最大化并行度，同时保证危险工具的权限交互不被打断 |
