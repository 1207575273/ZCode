### Step 5: 实现 src/commands/registry.ts

```typescript
// src/commands/registry.ts
import type { Command, CommandResult } from './types.js'

export class CommandRegistry {
  private readonly commands = new Map<string, Command>()

  register(cmd: Command): void {
    this.commands.set(cmd.name, cmd)
    for (const alias of cmd.aliases ?? []) {
      this.commands.set(alias, cmd)
    }
  }

  getAll(): Command[] {
    // 去重：只返回 name 匹配的（排除 alias 重复项）
    const seen = new Set<string>()
    return Array.from(this.commands.values()).filter(cmd => {
      if (seen.has(cmd.name)) return false
      seen.add(cmd.name)
      return true
    })
  }

  dispatch(input: string): CommandResult {
    if (!input.startsWith('/')) return { handled: false }

    const parts = input.slice(1).trim().split(/\s+/)
    const name = parts[0] ?? ''
    const args = parts.slice(1)

    const cmd = this.commands.get(name)
    if (!cmd) {
      return {
        handled: true,
        action: { type: 'error', message: `Unknown command: /${name}. Type /help for available commands.` },
      }
    }

    return cmd.execute(args)
  }
}
```

```typescript
import type { Command, CommandResult } from './types.js'
import type { CommandRegistry } from './registry.js'

export class HelpCommand implements Command {
  readonly name = 'help'
  readonly description = 'Show this help message'

  constructor(private readonly registry: CommandRegistry) {}

  execute(_args: string[]): CommandResult {
    const lines = ['Available commands:']
    for (const cmd of this.registry.getAll()) {
      const padding = ' '.repeat(Math.max(1, 14 - cmd.name.length))
      lines.push(`  /${cmd.name}${padding}${cmd.description}`)
    }
    return {
      handled: true,
      action: { type: 'show_help', content: lines.join('\n') },
    }
  }
}
```

```typescript
// src/commands/model.ts
import type { Command, CommandResult } from './types.js'

export class ModelCommand implements Command {
  readonly name = 'model'
  readonly aliases = ['m'] as const
  readonly description = 'Switch AI model interactively. /model <name> to switch directly. /model info for details.'

  constructor(
    private currentProvider: string,
    private currentModel: string,
  ) {}

  execute(args: string[]): CommandResult {
    const sub = args[0]

    if (!sub) {
      return { handled: true, action: { type: 'show_model_picker' } }
    }

    if (sub === 'info') {
      const content = [
        `Current model: ${this.currentModel}`,
        `Provider:      ${this.currentProvider}`,
        `Input price:   --`,
        `Output price:  --`,
      ].join('\n')
      return { handled: true, action: { type: 'show_help', content } }
    }

    // 直接指定模型名切换（provider 保持当前不变）
    return {
      handled: true,
      action: { type: 'switch_model', provider: this.currentProvider, model: sub },
    }
  }
}
```

```tsx
// src/ui/ModelPicker.tsx
import React, { useState } from 'react'
import { Box, Text, useInput } from 'ink'
import { useTerminalSize } from './useTerminalSize.js'
import { configManager } from '@config/config-manager.js'

interface ModelItem {
  provider: string
  model: string
}

interface ModelPickerProps {
  currentProvider: string
  currentModel: string
  onSelect: (provider: string, model: string) => void
  onCancel: () => void
}

function buildModelList(): ModelItem[] {
  const config = configManager.load()
  const items: ModelItem[] = []
  for (const [providerName, providerCfg] of Object.entries(config.providers)) {
    if (!providerCfg) continue
    for (const model of providerCfg.models) {
      items.push({ provider: providerName, model })
    }
  }
  return items
}

export function ModelPicker({ currentProvider, currentModel, onSelect, onCancel }: ModelPickerProps) {
  const models = buildModelList()
  const initialIndex = Math.max(
    0,
    models.findIndex(m => m.provider === currentProvider && m.model === currentModel),
  )
  const [selected, setSelected] = useState(initialIndex)
  const { columns } = useTerminalSize()

  useInput((_input, key) => {
    if (key.upArrow) setSelected(s => Math.max(0, s - 1))
    if (key.downArrow) setSelected(s => Math.min(models.length - 1, s + 1))
    if (key.return) {
      const item = models[selected]
      if (item) onSelect(item.provider, item.model)
    }
    if (key.escape) onCancel()
  })

  return (
    <Box flexDirection="column" width={columns}>
      <Box>
        <Text dimColor>{'─'.repeat(columns)}</Text>
      </Box>
      <Box paddingX={2} flexDirection="column">
        <Text bold>Switch model</Text>
        <Box marginY={1} flexDirection="column">
          {models.map((item, i) => {
            const isCurrent = item.provider === currentProvider && item.model === currentModel
            const isSelected = i === selected
            const prefix = isSelected ? '❯ ' : '  '
            const num = `${i + 1}.`
            const modelCol = item.model.padEnd(24)
            const providerCol = item.provider.padEnd(14)
            return (
              <Box key={`${item.provider}-${item.model}`} paddingLeft={1}>
                <Text color={isSelected ? 'cyan' : undefined}>
                  {prefix}{num.padEnd(4)}{modelCol}{providerCol}{'--  / --'}
                  {isCurrent ? ' ✓' : ''}
                </Text>
              </Box>
            )
          })}
        </Box>
        <Text dimColor>↑↓ navigate   Enter confirm   Esc cancel</Text>
      </Box>
    </Box>
  )
}
```

```tsx
// src/ui/App.tsx
import React, { useState, useMemo } from 'react'
import { Box, Text, useApp } from 'ink'
import { WelcomeScreen } from './WelcomeScreen.js'
import { ChatView } from './ChatView.js'
import { InputBar } from './InputBar.js'
import { PermissionDialog } from './PermissionDialog.js'
import { ModelPicker } from './ModelPicker.js'
import { useChat } from './useChat.js'
import { CommandRegistry } from '@commands/registry.js'
import { HelpCommand } from '@commands/help.js'
import { ClearCommand } from '@commands/clear.js'
import { ModelCommand } from '@commands/model.js'
import type { CommandAction } from '@commands/types.js'

interface AppProps {
  cwd?: string
}

export function App({ cwd = process.cwd() }: AppProps) {
  const { exit } = useApp()
  const {
    messages, streamingMessage, toolEvents, isStreaming, error,
    pendingPermission, resolvePermission,
    currentModel, currentProvider,
    submit, clearMessages, appendSystemMessage, switchModel,
  } = useChat()

  const [showModelPicker, setShowModelPicker] = useState(false)

  const registry = useMemo(() => {
    const reg = new CommandRegistry()
    const helpCmd = new HelpCommand(reg)
    reg.register(helpCmd)
    reg.register(new ClearCommand())
    reg.register(new ModelCommand(currentProvider, currentModel))
    return reg
  }, [currentProvider, currentModel])

  const handleCommandAction = (action: CommandAction | undefined) => {
    if (!action) return
    switch (action.type) {
      case 'clear_messages':
        clearMessages()
        break
      case 'show_help':
        appendSystemMessage(action.content)
        break
      case 'show_model_picker':
        setShowModelPicker(true)
        break
      case 'switch_model':
        switchModel(action.provider, action.model)
        appendSystemMessage(`✓ Switched to ${action.model} (${action.provider})`)
        break
      case 'error':
        appendSystemMessage(`Error: ${action.message}`)
        break
    }
  }

  const handleSubmit = (text: string) => {
    if (text === '/exit' || text === '/quit') {
      exit()
      return
    }
    if (text.startsWith('/')) {
      const result = registry.dispatch(text)
      if (result.handled) {
        handleCommandAction(result.action)
        return
      }
    }
    submit(text)
  }

  const started = messages.length > 0 || isStreaming

  return (
    <Box flexDirection="column" width="100%">
      {started ? (
        <ChatView messages={messages} streamingMessage={streamingMessage} toolEvents={toolEvents} />
      ) : (
        <WelcomeScreen model={currentModel} provider={currentProvider} cwd={cwd} />
      )}

      {error != null && (
        <Box paddingX={1}>
          <Text color="red">✗ {error}</Text>
        </Box>
      )}

      {pendingPermission != null ? (
        <PermissionDialog
          toolName={pendingPermission.toolName}
          args={pendingPermission.args}
          onResolve={resolvePermission}
        />
      ) : showModelPicker ? (
        <ModelPicker
          currentProvider={currentProvider}
          currentModel={currentModel}
          onSelect={(provider, model) => {
            switchModel(provider, model)
            appendSystemMessage(`✓ Switched to ${model} (${provider})`)
            setShowModelPicker(false)
          }}
          onCancel={() => setShowModelPicker(false)}
        />
      ) : (
        <InputBar onSubmit={handleSubmit} disabled={isStreaming} />
      )}
    </Box>
  )
}
```

**注意：** `AppProps` 移除了 `model` 和 `provider` 两个 props（已由 useChat 内部管理）。如果 `bin/zcli.ts` 传了这两个 prop，需要同步更新入口文件。先读取 `bin/zcli.ts`，按需修改。
